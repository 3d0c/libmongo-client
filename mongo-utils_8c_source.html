00060   guint8 *oid;

00062   gint32 tmp = GINT32_TO_BE (seq);
00063 
00064   if (machine_id == 0 || pid == 0)

00069   /* Sequence number, last 3 bytes
00070    * For simplicity's sake, we put this in first, and overwrite the
00071    * first byte later.
00072    */
00073   memcpy (oid + 4 + 2 + 2, &tmp, 4);
00074   /* First four bytes: the time, BE byte order */
00075   memcpy (oid, &t, 4);
00076   /* Machine ID, byte order doesn't matter, 3 bytes */
00077   memcpy (oid + 4, &machine_id, 3);
00078   /* PID, byte order doesn't matter, 2 bytes */
00079   memcpy (oid + 4 + 3, &pid, 2);
