00186 
00187       if (!mongo_util_parse_addr (addr, &host, &port))
00188         continue;

00190       nc = mongo_sync_connect (host, port, conn->slaveok);
00191       g_free (host);

00193         continue;
00194 
00195       nc = mongo_sync_reconnect (nc, force_master);
00196       e = errno;

00198       errno = e;

00203

00210         continue;
